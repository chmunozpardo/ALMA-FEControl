63  target.addCPDS();

65  }
66

68  const CartAssemblyConfig *cartAssy = feConfig -> getCartridgeConfig(pos);
69  if (cartAssy)
70  configureCartAssembly(*cartAssy, target);
71  }
72  LOG(LM_INFO) << "ConfigManager: Configured Front End (" << feConfig -> keyFacility_m << "," << feConfig -> keyFrontEnd_m << ")." << endl;
73  return true;
74  }
75

77  if (cartAssy) {
78  if (configureCartAssembly(*cartAssy, target)) {
79  LOG(LM_INFO) << "ConfigManager: Configured Cartridge Assembly ( port=" << cartAssy -> Id_m.port_m << ", band=" << cartAssy -> Id_m.band_m << " )." << endl;

81  }
82  }
