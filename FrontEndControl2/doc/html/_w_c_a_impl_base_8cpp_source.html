386 void WCAImplBase::monitorAction(Time *timestamp_p) {
387  // constant factor to convert 100 ns ticks to ms:
388  static const Time milliseconds = 10000;
389 
390  static const Time monitorInterval = 30 * milliseconds;
391  static const Time minMonitorInterval = 5000 * milliseconds;
392 
393  bool doMonitor = false;
394 
395  if (lastMonitorTime == 0 && logMonitors_m)
396  logMon(true);
397 
398  if (lastMonitorTime == 0 || (*timestamp_p - lastMonitorTime >= ((minimalMonitoring_m) ? minMonitorInterval : monitorInterval))) {
399  lastMonitorTime = *timestamp_p;
400  doMonitor = true;
401  }
402 
403  if (doMonitor) {
404  if (minimalMonitoring_m) {
405  pllUnlockDetectLatch_value = pllUnlockDetectLatch();
406  pllLockDetectVoltage_value = pllLockDetectVoltage();
407  pllCorrectionVoltage_value = pllCorrectionVoltage();
408  photomixerCurrent_value = photomixerCurrent();
409  pllRefTotalPower_value = pllRefTotalPower();
410  pllIfTotalPower_value = pllIfTotalPower();
411  pllAssemblyTemp_value = pllAssemblyTemp();
412  pllNullLoopIntegrator_value = pllNullLoopIntegrator();
413  } else {
414  switch (monitorPhase) {
415  case 0:
416  if (!executeNextMon()) {
417  if (randomizeAnalogMonitors_m)
418  randomizeMon();
419  monitorPhase = 1;
420  }
421  break;
422  case 1:
423  pllUnlockDetectLatch_value = pllUnlockDetectLatch();
424  ++monitorPhase;
425  break;
426  case 2:
427  ytoCoarseTune_value = ytoCoarseTune();
428  ++monitorPhase;
429  break;
430  case 3:
431  amcMultiplierDVoltage_value = amcMultiplierDVoltage();
432  ++monitorPhase;
433  break;
434  case 4:
435  photomixerEnable_value = photomixerEnable();
436  ++monitorPhase;
437  break;
438  case 5:
439  pllLoopBandwidthSelect_value = pllLoopBandwidthSelect();
440  ++monitorPhase;
441  break;
442  case 6:
443  pllSidebandLockSelect_value = pllSidebandLockSelect();
444  ++monitorPhase;
445  break;
446  case 7:
447  pllNullLoopIntegrator_value = pllNullLoopIntegrator();
448  ++monitorPhase;
449  //no break;
450  default:
451  if (logMonitors_m)
452  logMon();
453  monitorPhase = 0;
454  break;
455  }
456  }
457  }
458 }
