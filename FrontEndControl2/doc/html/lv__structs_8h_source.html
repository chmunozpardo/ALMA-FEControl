40 // 'field' is an index into offsets[] to finde where in data[] to find a value of type T.

42 template<typename T>
43 static T LVGet(int field, const unsigned char data[], const int offsets[], const T &) {
44  const T *target = reinterpret_cast<const T *>(data + offsets[field]);
45  if (target)
46  return *target;
47  else
48  return 0;
49 }

52 static T LVSet(int field, unsigned char data[], const int offsets[], const T &value) {
53  T *target = reinterpret_cast<T *>(data + offsets[field]);
54  if (target)
55  *target = value;
56 }
