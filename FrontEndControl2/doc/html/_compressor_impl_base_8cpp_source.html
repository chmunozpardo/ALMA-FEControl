140 unsigned long CompressorImplBase::getTimeSinceLastPowerOn() {
141  unsigned long target;
142  FEMC_ERROR feStatus = FEMC_NO_ERROR;
143  AmbRelativeAddr RCA(GET_TIME_SINCE_LAST_POWER_ON);
144  SYNCMON2(RCA, feStatus, target);
145  if (target >= 100000) // Compressor M&C module returns huge bogus value if the timer is not counting up anymore.
146  target = 99999; // sanitize.
147  LOG2_INT(FEMC_LOG_MONITOR, RCA, "COMPRESSOR:GET_TIME_SINCE_LAST_POWER_ON", feStatus, target);

177 void CompressorImplBase::monitorAction(Time *timestamp_p) {
178 
179  if (!timestamp_p)
180  return;
181 
182  // constant factor to convert 100 ns ticks to ms:
183  static const Time milliseconds = 10000;
184  static const Time monitorInterval = 10000 * milliseconds;
185  bool doMonitor = false;
186 
187  if (lastMonitorTime == 0 || (*timestamp_p - lastMonitorTime >= monitorInterval)) {
188  lastMonitorTime = *timestamp_p;
189  doMonitor = true;
190  }
191 
192  static bool monitorOnce(true);
193  if (monitorOnce) {
194  monitorOnce = false;
195  getECUType_value = getECUType();
196  getSWRevisionLevel_value = getSWRevisionLevel();
197  }
198 
199  if (doMonitor) {
200  LOG(LM_TRACE) << "CompressorImplBase::monitorAction" << endl;
201  getTemp1_value = getTemp1();
202  getTemp2_value = getTemp2();
203  getTemp3_value = getTemp3();
204  getTemp4_value = getTemp4();
205  getReturnLinePressure_value = getReturnLinePressure();
206  getAuxTransducer_value = getAuxTransducer();
207  getSupplyPressure_value = getSupplyPressure();
208  getPressureAlarm_value = getPressureAlarm();
209  getTempAlarm_value = getTempAlarm();
210  getDriveIndicator_value = getDriveIndicator();
211  getICCUStatusError_value = getICCUStatusError();
212  getICCUCableError_value = getICCUCableError();
213  getFETIMStatusError_value = getFETIMStatusError();
214  getFETIMCableError_value = getFETIMCableError();
215  getInterlockOverride_value = getInterlockOverride();
216  getFaultStatusError_value = getFaultStatusError();
217  getTimeSinceLastPowerOn_value = getTimeSinceLastPowerOn();
218  getTimeSinceLastPowerOff_value = getTimeSinceLastPowerOff();
219  }
220 }
