79 bool LORTMImpl::PhaseLockTuningInit(short band, double freqLO1, double freqLO2, double freqFLOOG, short sbLock) {
80 
81  // check parameters:
82  if (band < 1 || band > 10) {
83  LOG(LM_ERROR) << "LORTMImpl::PhaseLockTuningInit band out of range" << endl;
84  return false;
85  }
86 
87  // divide the LO frequencies by the cartridge cold multiplier to get the LORTM output frequencies:
88  double coldMult = (double) ColdCartImpl::getMultiplier(band);
89  // min and max LORTM sideband frequencies possible:
90  const double freqMinSb = 24.0;
91  const double freqMaxSb = 150.0;
92  // min and max LO frequencies possible:
93  double minLO = freqMinSb * coldMult;
94  double maxLO = freqMaxSb * coldMult;
95 
96  // force frequencies into allowed range or equal to zero:
97  bool zeroLO1 = false;
98  if (freqLO1 < minLO) {
99  freqLO1 = 0;
100  zeroLO1 = true;
101  } else if (freqLO1 > maxLO) {
102  LOG(LM_ERROR) << "LORTMImpl::PhaseLockTuningInit freqLO1=" << freqLO1 << " out of range. Set to" << maxLO << endl;
103  freqLO1 = maxLO;
104  }
105  bool zeroLO2 = false;
106  if (freqLO2 < minLO) {
107  freqLO2 = 0;
108  zeroLO2 = true;
109  } else if (freqLO2 > maxLO) {
110  LOG(LM_ERROR) << "LORTMImpl::PhaseLockTuningInit freqLO2=" << freqLO2 << " out of range. Set to" << maxLO << endl;
111  freqLO2 = maxLO;
112  }
113  // generate error message if both LO frequencies were forced to zero:
114  if (zeroLO1 && zeroLO2)
115  LOG(LM_ERROR) << "LORTMImpl::PhaseLockTuningInit freqLO1 and freqLO2 were both below range and set to zero." << endl;
116 
117  // Start assigning variables to object state:
118  resetTuning();
119  band_m = band;
120  freqFLOOG_m = freqFLOOG;
121  sbLock_m = sbLock;
122 
123  // Assign freqLSB and freqUSB so that they are in order or that freqUSB is zero:
124  double freq1 = freqLO1 / coldMult;
125  double freq2 = freqLO2 / coldMult;
126  if (freq1 == 0.0) {
127  if (freq2 == 0.0) {
128  // error message is printed above.
129  return false;
130  } else {
131  freqLSB_m = freq2;
132  freqUSB_m = 0.0;
133  }
134  } else {
135  if (freq2 == 0.0) {
136  freqLSB_m = freq1;
137  freqUSB_m = 0.0;
138  } else if (freq1 < freq2) {
139  freqLSB_m = freq1;
140  freqUSB_m = freq2;
141  } else {
142  freqLSB_m = freq2;
143  freqUSB_m = freq1;
144  }
145  }
146 
147  // add the FLOOG offset:
148  if (freqLSB_m != 0.0)
149  freqLSB_m += ((sbLock_m == 0) ? freqFLOOG_m : -freqFLOOG_m);
150  if (freqUSB_m != 0.0)
151  freqUSB_m += ((sbLock_m == 0) ? freqFLOOG_m : -freqFLOOG_m);
152 
153  // get the Ref2 frequency divide-by value depending on the band under test:
154  double ref2DivideBy = getRef2DivideBy(band);
155 
156  if (freqUSB_m == 0.0)
157  freqRef2_m = 0.0;
158  else
159  // set the ref2 frequency based on the distance between the requested frequencies and the divide-by value:
160  freqRef2_m = ((freqUSB_m * coldMult) - (freqLSB_m * coldMult)) / ref2DivideBy;
161 
162  LOG(LM_INFO) << "LORTMImpl::PhaseLockTuningInit(band=" << band_m << " freqLO1=" << freqLO1 << " freqLO2=" << freqLO2
163  << ") freqLSB=" << freqLSB_m << " freqUSB=" << freqUSB_m << " ref2DivideBy=" << ref2DivideBy << endl;
164 
165  //freqRefLaser_m = 192642.675; // GHz (from LORTM#4 documentation)
166  const double freqRef3 = 0.125; // GHz
167 
168  double freqBeatNote = freqLSB_m + freqRef2_m; // GHz
169  freqSlaveLaser_m = freqRefLaser_m - freqBeatNote; // GHz
170  unsigned long freqSlaveMHz = (unsigned long) (freqSlaveLaser_m * 1000.0);
171 
172  bool oddMOnly = false;
173 
174  LOG(LM_INFO) << "LORTMImpl::PhaseLockTuningInit "
175  << " freqRef2=" << freqRef2_m << ", oddMOnly=" << oddMOnly
176  << " freqRefLaser=" << setprecision(9) << freqRefLaser_m << setprecision(6)
177  << " freqRef3=" << freqRef3 << " freqBeatNote=" << freqBeatNote
178  << " freqSlaveLaser=" << setprecision(9) << freqSlaveLaser_m << setprecision(6)
179  << " freqSlaveMHz=" << freqSlaveMHz << endl;
180 
181  bool success = false;
182  if (!oddMOnly) {
183  if (!success) {
184  // try M=2 case: M*Fref1-Fref3 <= 39 GHz
185  factorM_m = 2;
186  freqRef1_m = (freqBeatNote + freqRef3) / ((double) factorM_m);
187  double freqTest = ((double) factorM_m) * freqRef1_m - freqRef3;
188  if (freqTest <= 39.0)
189  success = true;
190  LOG(LM_INFO) << "PhaseLockTuningInit try M=2 freqRef1=" << freqRef1_m
191  << " freqTest=" << freqTest << " success=" << success << endl;
192  }
193  }
194  if (!success) {
195  // try M=5 case: 64.5 GHz <= M*Fref1-Fref3 < 100 GHz
196  factorM_m = 5;
197  freqRef1_m = (freqBeatNote + freqRef3) / ((double) factorM_m);
198  double freqTest = ((double) factorM_m) * freqRef1_m - freqRef3;
199  if (freqTest >= 64.5 && freqTest < 100.0)
200  success = true;
201  LOG(LM_INFO) << "PhaseLockTuningInit try M=5 freqRef1=" << freqRef1_m
202  << " freqTest=" << freqTest << " success=" << success << endl;
203  }
204  if (!success) {
205  // try M=7 case: 100 GHz <= M*Fref1-Fref3 <= 150 GHz
206  factorM_m = 7;
207  freqRef1_m = (freqBeatNote + freqRef3) / ((double) factorM_m);
208  double freqTest = ((double) factorM_m) * freqRef1_m - freqRef3;
209  if (freqTest >= 100.0 && freqTest <= 150.0)
210  success = true;
211  LOG(LM_INFO) << "PhaseLockTuningInit try M=7 freqRef1=" << freqRef1_m
212  << " freqTest=" << freqTest << " success=" << success << endl;
213  }
214  if (success)
215  LORTMImplBase::PhaseLockTuningInit(freqSlaveMHz);
216  else
217  resetTuning();
218  return success;
219 }

321 double LORTMImpl::getRef2DivideBy(short band) const {
322  // get the Ref2 frequency divide-by value depending on the band under test:
323  // switch pos refers to the switches in the phase electronics to select the divider.
324  //
325  // This is set based on the cold multiplication factor, not the WCA.
326  // The divide-by factor should be 2x the cold multiplier.
327  //
328 
329  return 2.0 * ((double) ColdCartImpl::getMultiplier(band));
330
