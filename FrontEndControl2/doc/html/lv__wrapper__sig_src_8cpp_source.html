60 DLLEXPORT short sigSrcControlInit() {
61  ++connectedSigSrcClients;
62  if (sigSrc && sigSrcValid) {
63  LOG(LM_INFO) << "sigSrcControlInit: connectedSigSrcClients=" << connectedSigSrcClients << endl;
64  return 0;
65  }
66  if (LVWrapperInit() < 0) {
67  LOG(LM_ERROR) << "sigSrcControlInit: LVWrapperInit failed." << endl;
68  return -1;
69  }
70  LOG(LM_INFO) << "sigSrcControlInit: connectedSigSrcClients=" << connectedSigSrcClients << endl;
71  try {
72  // get the path where the FrontendControlDLL.ini file is located:
73  string iniPath, tmp;
74  splitPath(iniFileName, iniPath, tmp);
75  if (iniPath.empty())
76  iniPath = ".";
77 
78  // open the FrontendControlDLL.ini file:
79  CIniFile configINI(iniFileName);
80  configINI.ReadFile();
81 
82  // look for the item specifying a separate file for signal source config:
83  tmp = configINI.GetValue("configFiles", "signalSource");
84  if (tmp.empty())
85  // not found. use the FrontendControlDLL.ini file to load signal source config:
86  sigSrcIniFilename = iniFileName;
87  else {
88  // if found, open that file to load the signal source config data from:
89  sigSrcIniFilename = iniPath + "/" + tmp;
90  LOG(LM_INFO) << "Using signal source configuration file '" << sigSrcIniFilename << "'" << endl;
91  configINI.Clear();
92  configINI.SetPath(sigSrcIniFilename);
93  if (!configINI.ReadFile()) {
94  LOG(LM_ERROR) << "Error reading signal source config file." << endl;
95  return -1;
96  }
97  }
98 
99  tmp = configINI.GetValue("connection", "channelSigSrc");
100  if (!tmp.empty())
101  CANChannelSigSrc = from_string<unsigned long>(tmp);
102  tmp = configINI.GetValue("connection", "nodeAddressSigSrc");
103  if (!tmp.empty()) {
104  if (tmp[0] == '0' && (tmp[1] == 'x' || tmp[1] == 'X'))
105  tmp = tmp.substr(2);
106  nodeAddressSigSrc = from_string<unsigned long>(tmp, std::hex);
107  }
108  LOG(LM_INFO) << "Connecting to signal source on CAN" << CANChannelSigSrc
109  << " nodeAddress=" << uppercase << hex << setw(2) << setfill('0') << nodeAddressSigSrc << dec << setw(0) << endl;
110 
111  // get the facility code from the new key facilityId:
112  tmp = configINI.GetValue("configuration", "facilityIdSigSrc");
113  if (!tmp.empty())
114  facilityIdSigSrc = from_string<unsigned long>(tmp);
115  else {
116  // but fall back on the old name if that doesn't work:
117  tmp = configINI.GetValue("configuration", "providerCodeSigSrc");
118  if (!tmp.empty())
119  facilityIdSigSrc = from_string<unsigned long>(tmp);
120  }
121 
122  tmp = configINI.GetValue("configuration", "configIdSigSrc");
123  if (!tmp.empty())
124  configIdSigSrc = from_string<unsigned long>(tmp);
125  LOG(LM_INFO) << "Using configuration facilityIdSigSrc=" << facilityIdSigSrc << " configIdSigSrc=" << configIdSigSrc << endl;
126 
127  } catch (...) {
128  LOG(LM_ERROR) << "sigSrcControlInit exception loading configuration file." << endl;
129  FEMCEventQueue::addStatusMessage(false, "An exception occurred while loading the configuration file.");
130  return -1;
131  }
132 
133  short ret = -1;
134  if (nodeAddressSigSrc) {
135  sigSrc = new SignalSourceImpl(CANChannelSigSrc, nodeAddressSigSrc, "ESN-SigSrc");
136  if (sigSrc -> connected()) {
137  ret = 0;
138 
139  } else {
140  LOG(LM_ERROR) << "sigSrcControlInit failed to connect to signal source control module." << endl;
141  FEMCEventQueue::addStatusMessage(false, "Failed to connect to signal source control module.");
142  }
143  }
144 
145  // If we are in a debugging mode where no CAN data is sent, pretend we got connected anyway:
146  if (CAN_noTransmit)
147  ret = 0;
148 
149  if (ret == 0)
150  ret = sigSrcLoadConfiguration((short) configIdSigSrc);
151 
152  return ret;
153 }

191 DLLEXPORT short sigSrcGetNextConfiguration(short reset, short *configId, char *description) {
192  static vector<Configuration::Record> configsList;
193  static vector<Configuration::Record>::const_iterator nextConfig = configsList.begin();
194  bool ret = true;
195 
196  if (!configId)
197  LOG(LM_ERROR) << "sigSrcGetNextConfiguration: configId is NULL." << endl;
198  if (!description)
199  LOG(LM_ERROR) << "sigSrcGetNextConfiguration: description is NULL." << endl;
200 
201  *configId = 0;
202  *description = 0x00;
203 
204  if (reset) {
205  ConfigProvider *provider = new ConfigProviderIniFile(sigSrcIniFilename);
206  ret = provider -> getAllConfigurations(facilityIdSigSrc, configsList);
207  nextConfig = configsList.begin();
208  delete provider;
209  } else
210  nextConfig++;
211 
212  if (!ret)
213  LOG(LM_ERROR) << "sigSrcGetNextConfiguration: got no configs." << endl;
214 
215  if (ret) {
216  if (nextConfig == configsList.end()) {
217  ret = false;
218  } else {
219  *configId = (*nextConfig).configId_m;
220  strcpy(description, (*nextConfig).description_m.c_str());
221  LOG(LM_INFO) << "sigSrcGetNextConfiguration: " << description << endl;
222  ret = true;
223  }
224  }
225  return ret ? 0 : -1;
226 }

228 DLLEXPORT short sigSrcLoadConfiguration(short configId) {
229 
230  if (configId <= 0)
231  configId = configIdSigSrc;
232 
233  ConfigProvider *provider(NULL);
234  provider = new ConfigProviderIniFile(sigSrcIniFilename);
235 
236  ConfigManager configMgr(*provider);
237  Configuration config(facilityIdSigSrc, configId);
238  if (!config.load(*provider))
239  return -1;
240 
241  delete provider;
242 
243  const CartAssemblyConfig *caConfig = config.getCartAssemblyConfig();
244  if (caConfig) {
245  const WCAConfig &wcaConfig = caConfig -> WCA_m;
246  int wcaBand = wcaConfig.band_m;
247  char name[30];
248  int port(caConfig -> Id_m.port_m);
249  if (port < 1 || port > 10) {
250  LOG(LM_ERROR) << "sigSrcLoadConfiguration error: out of range port=" << port << endl;
251  return -1;
252 
253  } else if (wcaBand < 1 || wcaBand > 10) {
254  LOG(LM_ERROR) << "sigSrcLoadConfiguration error: out of range band=" << wcaBand << endl;
255  return -1;
256 
257  } else {
258  LOG(LM_INFO) << "sigSrcLoadConfiguration: port=" << port << " wcaBand=" << wcaBand << endl;
259  sprintf(name, "band %d Signal Source", wcaBand);
260  if (sigSrcWCA_p) {
261  sigSrc -> deleteCartridge();
262  WHACK(sigSrcWCA_p);
263  }
264  sigSrcWCA_p = new WCAImpl(CANChannelSigSrc, nodeAddressSigSrc, name, port, wcaBand, wcaConfig.ESN_m);
265  sigSrc -> addCartridge(port, *sigSrcWCA_p);
266  }
267  }
268  configMgr.configure(config, *sigSrc);
269  sigSrcValid = true;
270  LOG(LM_INFO) << "sigSrcLoadConfiguration successful" << endl;
271  return 0;
272 }
