149 void LPRImplBase::monitorAction(Time *timestamp_p) {
150 
151  if (!timestamp_p)
152  return;
153 
154  // constant factor to convert 100 ns ticks to ms:
155  static const Time milliseconds = 10000;
156  static const Time monitorInterval = 150 * milliseconds;
157 
158  bool doMonitor = false;
159 
160  if (lastMonitorTime == 0 && logMonitors_m)
161  logMon(true);
162 
163  if (lastMonitorTime == 0 || (*timestamp_p - lastMonitorTime >= monitorInterval)) {
164  lastMonitorTime = *timestamp_p;
165  doMonitor = true;
166  }
167 
168  if (doMonitor) {
169  switch (monitorPhase) {
170  case 0:
171  if (!executeNextMon()) {
172  if (randomizeAnalogMonitors_m)
173  randomizeMon();
174  monitorPhase = 1;
175  }
176  break;
177  case 1:
178  opticalSwitchPort_value = opticalSwitchPort();
179  ++monitorPhase;
180  break;
181  case 2:
182  opticalSwitchShutter_value = opticalSwitchShutter();
183  ++monitorPhase;
184  break;
185  case 3:
186  opticalSwitchState_value = opticalSwitchState();
187  ++monitorPhase;
188  break;
189  case 4:
190  opticalSwitchBusy_value = opticalSwitchBusy();
191  ++monitorPhase;
192  // no break;
193  default:
194  if (logMonitors_m)
195  logMon();
196  monitorPhase = 0;
197  break;
198  };
199  }
200 }
