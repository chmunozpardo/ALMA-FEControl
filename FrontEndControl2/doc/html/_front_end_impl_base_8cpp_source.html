119 bool FrontEndImplBase::powerEnableModule(int port) {
120  if (port >= 1 && port <= 10) {
121  AmbRelativeAddr RCA(powerEnableModule_RCA);
122  RCA |= ((port - 1) << 4);
123  unsigned char val(0);
124  sem_t synchLock;
125  sem_init(&synchLock, 0, 0);
126  FEMC_ERROR status = syncMonitor(RCA, val, synchLock);
127  sem_destroy(&synchLock);
128  getLogger().log(FEMC_LOG_MONITOR, "POWER_ENABLE_MODULE", RCA, (signed char) status, val, 0.0);
129  if (status == FEMC_NO_ERROR && val != 0)
130  return true;
131  }
132  return false;
133 }

197 void FrontEndImplBase::monitorAction(Time *timestamp_p) {
198 
199  if (!timestamp_p)
200  return;
201 
202  // constant factor to convert 100 ns ticks to ms:
203  static const Time milliseconds = 10000;
204  static const Time monitorInterval = 5000 * milliseconds;
205 
206  bool doMonitor = false;
207 
208  if (lastMonitorTime == 0 || (*timestamp_p - lastMonitorTime >= monitorInterval)) {
209  lastMonitorTime = *timestamp_p;
210  doMonitor = true;
211  }
212 
213  if (doMonitor) {
214  LOG(LM_TRACE) << "FrontEndImplBase::monitorAction" << endl;
215  AMBSINumErrors();
216  AMBSINumTransactions();
217  AMBSITemperature();
218  numEnabledModules_value = numEnabledModules();
219  }
220 }
