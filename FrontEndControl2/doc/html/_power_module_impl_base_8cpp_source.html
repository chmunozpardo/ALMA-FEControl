154 void PowerModuleImplBase::monitorAction(Time *timestamp_p) {
155  if (!timestamp_p)
156  return;
157 
158  // constant factor to convert 100 ns ticks to ms:
159  static const Time milliseconds = 10000;
160  static const Time monitorInterval = 2000 * milliseconds;
161 
162  bool doMonitor = false;
163 
164  if (lastMonitorTime == 0 && logMonitors_m)
165  logMon(true);
166 
167  if (lastMonitorTime == 0 || (*timestamp_p - lastMonitorTime >= monitorInterval)) {
168  lastMonitorTime = *timestamp_p;
169  doMonitor = true;
170  }
171 
172  if (doMonitor) {
173  enableModule_value = enableModule();
174  while (executeNextMon()) ;
175  if (randomizeAnalogMonitors_m)
176  randomizeMon();
177  if (logMonitors_m)
178  logMon();
179  }
180 }
