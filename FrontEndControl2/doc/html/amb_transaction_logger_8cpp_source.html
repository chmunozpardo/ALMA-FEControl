144 int AmbTransactionLogger::LogBuffer::getEntriesToLog(LogBuffer_t &toLog) {
145  // get the range of indexes to return:
146  int nextLogBegin, nextLogEnd, maxCount;
147  maxCount = getNextLogIndex(nextLogBegin, nextLogEnd);
148  // quit if nothing to log:
149  if (maxCount == 0)
150  return 0;
151  // prepare the output buffer:
152  toLog.clear();
153  toLog.reserve(maxCount);
154  // move up to maxCount items from buffer_m into toLog:
155  // Stop early if a NULL pointer is encountered.
156  int pos(nextLogBegin), count(0);
157  while (count < maxCount && buffer_m[pos] != NULL) {
158  ++count;
159  // ownership of the pointer is transferred over to toLog:
160  toLog.push_back(buffer_m[pos]);
161  buffer_m[pos] = NULL;
162  ++pos;
163  if (pos == bufferSize_m)
164  pos = 0;
165  }
166  // Save the next starting log index:
167  setNextLogIndex(pos, count);
168  // return the number of items actually moved:
169  return count;
170 }

186 int AmbTransactionLogger::LogBuffer::getNextInsertIndex() {
187  pthread_mutex_lock(&mutex);
188  // get the next insert position:
189  int preInsertPos(nextInsertPos_m);
190  int preLogPos(nextLogPos_m);
191  int preNumItems(numItems_m);
192  int pos(nextInsertPos_m);
193  // if an entry already exists then the buffer is full; return -1:
194  if (buffer_m[pos] != NULL)
195  pos = -1;
196  // otherwise, increment and account for circular buffer wraparound:
197  else {
198  ++numItems_m;
199  ++nextInsertPos_m;
200  if (nextInsertPos_m == bufferSize_m)
201  nextInsertPos_m = 0;
202  }
203  int postInsertPos(nextInsertPos_m);
204  int postNumItems(numItems_m);
205  // unlock and return the saved position:
206  pthread_mutex_unlock(&mutex);
207  //LOG(LM_DEBUG) << "LogBuffer::getNextInsertIndex preInsertPos=" << preInsertPos
208  // << " preNumItems=" << preNumItems << " preLogPos=" << preLogPos
209  // << " postInsertPos=" << postInsertPos
210  // << " postNumItems=" << postNumItems << " pos=" << pos << endl;
211  return pos;
212 }

325  if (!owner)
326  pthread_exit(NULL);
327 
328  LogBuffer_t toLog;
329  LogEntry *entry;
330 
331  while (true) {

334  entry = *it;
