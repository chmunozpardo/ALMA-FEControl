211  FEMC_ERROR syncMonitor(AmbRelativeAddr RCA, T &target, sem_t &synchLock) {
212  FEMC_ERROR ret(FEMC_NO_ERROR);
213  AmbErrorCode_t status(AMBERR_NOERR);
214  AmbDataLength_t dataLength;
215  AmbDataMem_t data[8];
216  Time timestamp;
217  // send monitor request with the semaphore:
218  monitor(RCA, dataLength, data, &synchLock, &timestamp, &status);
219  // wait on the semaphore:
220  sem_wait(&synchLock);
221  // check for errors and return result:
222  if (status == AMBERR_NOERR)
223  ret = unpack(target, dataLength, data);
224  else {
225  ret = FEMC_AMB_ERROR;
226  if (logAmbErrors_m) {
227  LOG(LM_ERROR) << "FEHardwareDevice: AMB error=" << status << " RCA=" << " 0x" << std::uppercase << std::hex << std::setw(6) << std::setfill('0') << RCA << std::endl;
228  }
229  }
230  if (!monitorIgnorableError(ret))
231  ++errorCount; // increment error counter.
232  return ret;
233  }

236  template<typename T>
237  FEMC_ERROR syncMonitorWithRetry(AmbRelativeAddr RCA, T &target, int retries = 12) {
238  FEMC_ERROR ret(FEMC_NO_ERROR);
239  sem_t synchLock;
240  sem_init(&synchLock, 0, 0);
241  bool done = false;
242  while (!done && retries > 0) {
243  ret = syncMonitor(RCA, target, synchLock);
244  // if AMB error, quit without retrying:
245  if (ret == FEMC_AMB_ERROR)
246  done = true;
247  // no AMB error. Check for retry required:
248  else if (ret == FEMC_HARDW_RETRY_WARN)
249  --retries;
250  // No error, or some other error than what we checked for - return:
251  else
252  done = true;
253  }
254  sem_destroy(&synchLock);
255  return ret;
256  }

266  FEMC_ERROR syncCommand(AmbRelativeAddr RCA, T value) {
267  FEMC_ERROR ret(FEMC_NO_ERROR);
268  AmbErrorCode_t status(AMBERR_NOERR);
269  AmbDataLength_t dataLength;
270  AmbDataMem_t data[8];
271  pack(value, dataLength, data);
272  Time timestamp;
273  // send command with a semaphore:
274  sem_t synchLock;
275  sem_init(&synchLock, 0, 0);
276  command(RCA, dataLength, data, &synchLock, &timestamp, &status);
277  // wait on the semaphore (hence synchronous):
278  sem_wait(&synchLock);
279  // send a monitor request to the same RCA with the semaphore and wait:
280  monitor(RCA, dataLength, data, &synchLock, NULL, NULL);
281  sem_wait(&synchLock);
282  sem_destroy(&synchLock);
283  // unpack the resulting FEStatus, discarding the rest of the payload:
284  T temp;
285  ret = unpack(temp, dataLength, data);

324 // These macros support declaring and defining a MONITORS_REGISTRY in a derived class having the interface:
325 // addMon(float *target, (pointer to member function) pf);
326 // - add a monitor point and its corresponding function to the registry.
327 // randomizeMon()
328 // - randomize the order of monitor points in the registry.
329 // executeNextMon()
330 // - call a monitor function from the registry and put its value in the target float variable.
331 
332 #define DECLARE_MONITORS_REGISTRY(CLASS) \
333  typedef float (CLASS ::*MonFuncPtr) (void); \
334  typedef std::pair<float *, MonFuncPtr> RegEntry; \
335  std::vector<RegEntry> monitorRegistry; \
336  std::vector<RegEntry>::iterator nextMon; \
337  void addMon(float *target, MonFuncPtr pf) \
338  { monitorRegistry.push_back(RegEntry(target, pf)); } \
339  void randomizeMon() \
340  { std::random_shuffle(monitorRegistry.begin(), monitorRegistry.end()); } \
341  bool executeNextMon();
342 
343 #define DEFINE_MONITORS_REGISTRY(CLASS) \
344  bool CLASS ::executeNextMon() { \
345  if (nextMon == monitorRegistry.end()) { \
346  nextMon = monitorRegistry.begin(); \
347  return false; \
348  } else { \
349  RegEntry ent = *nextMon; \
350  float val = (this ->* ent.second)(); \
351  if (ent.first) *(ent.first) = val; \
352  ++nextMon; return true; \
353  }}
