236 void FETIMImplBase::monitorAction(Time *timestamp_p) {
237 
238  if (!timestamp_p)
239  return;
240 
241  // constant factor to convert 100 ns ticks to ms:
242  static const Time milliseconds = 10000;
243  static const Time monitorInterval = 20 * milliseconds;
244 
245  bool doMonitor = false;
246 
247  if (lastMonitorTime == 0 && logMonitors_m)
248  logMon(true);
249 
250  if (lastMonitorTime == 0 || (*timestamp_p - lastMonitorTime >= monitorInterval)) {
251  lastMonitorTime = *timestamp_p;
252  doMonitor = true;
253  }
254 
255  if (doMonitor) {
256  switch (monitorPhase) {
257  case 0:
258  if (!executeNextMon()) {
259  if (randomizeAnalogMonitors_m)
260  randomizeMon();
261  monitorPhase = 1;
262  }
263  break;
264  case 1:
265  internalTemperatureOOR_value = internalTemperatureOOR();
266  ++monitorPhase;
267  break;
268  case 2:
269  externalTemperatureOOR1_value = externalTemperatureOOR1();
270  ++monitorPhase;
271  break;
272  case 3:
273  externalTemperatureOOR2_value = externalTemperatureOOR2();
274  ++monitorPhase;
275  break;
276  case 4:
277  airflowSensorOOR_value = airflowSensorOOR();
278  ++monitorPhase;
279  break;
280  case 5:
281  heliumBufferPressureOOR_value = heliumBufferPressureOOR();
282  ++monitorPhase;
283  break;
284  case 6:
285  sensorSingleFailed_value = sensorSingleFailed();
286  ++monitorPhase;
287  break;
288  case 7:
289  sensorMultiFailed_value = sensorMultiFailed();
290  ++monitorPhase;
291  break;
292  case 8:
293  glitchCounterTriggered_value = glitchCounterTriggered();
294  ++monitorPhase;
295  break;
296  case 9:
297  delayShutdownTriggered_value = delayShutdownTriggered();
298  ++monitorPhase;
299  break;
300  case 10:
301  finalShutdownTriggered_value = finalShutdownTriggered();
302  ++monitorPhase;
303  break;
304  case 11:
305  getCompressorStatus_value = getCompressorStatus();
306  ++monitorPhase;
307  break;
308  case 12:
309  getCompressorInterlockStatus_value = getCompressorInterlockStatus();
310  ++monitorPhase;
311  break;
312  case 13:
313  getCompressorCableStatus_value = getCompressorCableStatus();
314  ++monitorPhase;
315  // no break;
316  default:
317  if (logMonitors_m)
318  logMon();
319  monitorPhase = 0;
320  break;
321  };
322  }
323 }
