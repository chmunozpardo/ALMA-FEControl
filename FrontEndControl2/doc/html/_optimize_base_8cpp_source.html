72 bool OptimizeBase::stopWorkerThread() {
73  if (!data_m.start_m)
74  return true;
75  if (data_m.stopped_m)
76  return true;
77  LOG(LM_INFO) << "OptimizeBase(" << name_m << "): stopping worker thread..." << endl;
78  int retry = 250;
79  data_m.stop_m = true;
80  do {
81  SLEEP(100);
82  } while (!data_m.stopped_m && retry--);
83  if (!data_m.stopped_m) {
84  int elapsed = (250 - retry) / 10;
85  LOG(LM_INFO) << "OptimizeBase(" << name_m << "): NOT stopped after " << elapsed << " seconds." << endl;
86  }
87  data_m.reset();
88  return data_m.stopped_m;
89 }

91 void *OptimizeBase::optimizeThread(OptimizeBase *owner) {
92  // refuse to run if not passed an owner pointer:
93  if (!owner)
94  return NULL;
95 
96  optimizeData &data_m = owner -> data_m;
97  // TESTING: try caching the name pointer to avoid access violation on exit.
98  std::string name(owner -> name_m);
99 
100  bool done = false;
101  while (!done) {
102  // if told to stop exit the loop:
103  if (data_m.stop_m)
104  done = true;
105  else {
106  // If not stopped, call the derived class' optimizeAction:
107  owner -> optimizeAction();
108  // If the optimization is done, stop:
109  if (owner -> finished()) {
110  LOG(LM_INFO) << "OptimizeBase(" << name << "): optimizeAction finished." << endl;
111  done = true;
112  }
113  }
114  }
115  // call the exitAction with the result indicating successful finish.
116  owner -> exitAction(owner -> success());
117 
118  // time to stop; acknowledge with stopped_m and exit the thread:
119  LOG(LM_INFO) << "OptimizeBase(" << name << "): exiting worker thread." << endl;
120  pthread_mutex_lock(&data_m.lock_m);
121  data_m.stopped_m = true;
122  pthread_mutex_unlock(&data_m.lock_m);
123  return NULL;
124 }
