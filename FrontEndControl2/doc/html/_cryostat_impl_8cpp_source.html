45 bool CryostatImpl::cryoPumpingEnable(bool val) {
46  if (val) {
47  if (!backingPumpEnable()) {
48  LOG(LM_INFO) << "CryostatImpl::cryoPumpingEnable(true): backing pump is not enabled." << endl;
49  return false;
50  }
51  LOG(LM_INFO) << "CryostatImpl::cryoPumpingEnable(true): starting cryostat pumping process..." << endl;
52  if (!cryoPumpMonitor_mp)
53  cryoPumpMonitor_mp = new CryostatPumping(*this);
54  cryoPumpMonitor_mp -> start();
55  } else if (cryoPumpMonitor_mp) {
56  LOG(LM_INFO) << "CryostatImpl::cryoPumpingEnable(false): stopping cryostat pumping process..." << endl;
57  cryoPumpMonitor_mp -> stop();
58  }
59  return true;
60 }
