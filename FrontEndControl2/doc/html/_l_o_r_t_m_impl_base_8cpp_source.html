104  Time timestamp;
105  AmbErrorCode_t status = AMBERR_NOERR;

205 unsigned short LORTMImplBase::SystemGetStatus() {
206  SystemGetStatus_value = 0;
207  AmbDataLength_t dataLength;
208  AmbDataMem_t data[8];
209  sem_t synchLock;
210  Time timestamp;
211  AmbErrorCode_t status = AMBERR_NOERR;
212  FEMC_ERROR feStatus = FEMC_NO_ERROR;
213  sem_init(&synchLock, 0, 0);
214  monitor(SYSTEM_GET_STATUS, dataLength, data, &synchLock, &timestamp, &status);
215  sem_wait(&synchLock);

221  systemWarning = (SystemGetStatus_value & 0x0004) != 0;

225  }
226  } else
227  feStatus = FEMC_UNPACK_ERROR;
228  getLogger().log(FEMC_LOG_MONITOR, "LORTM:SYSTEM_GET_STATUS", SYSTEM_GET_STATUS, feStatus, SystemGetStatus_value, 0.0);
229  return SystemGetStatus_value;
