144 void CartAssembly::measureSISVoltageError() {
145  if (band_m <= 2) {
146  // No SIS mixers in band 1 and 2. Nothing to do.
147  return;
148  }
149  if (coldCart_mp) {
150  // pause CCA monitoring while measuring voltage errors:
151  pauseMonitor(true, true, "measureSISVoltageError");
152 
153  // If band 5 or above, save the prior enabled state of the magnets:
154  float iSet01(0.0), iSet02(0.0), iSet11(0.0), iSet12(0.0);
155  bool magnetPrior(false);
156 
157  if (band_m <= 4)
158  // delay to allow cartridge with no magnets to fully bias up:
159  SLEEP(1500);
160 
161  else {
162  magnetPrior = coldCart_mp -> getSISMagnetEnableSetting();
163 
164  // save the prior current settings voltages, if enabled:
165  if (magnetPrior) {
166  iSet01 = coldCart_mp -> getSISMagnetCurrentSetting(0, 1);
167  iSet02 = coldCart_mp -> getSISMagnetCurrentSetting(0, 2);
168  iSet11 = coldCart_mp -> getSISMagnetCurrentSetting(1, 1);
169  iSet12 = coldCart_mp -> getSISMagnetCurrentSetting(1, 2);
170  }
171 
172  // enable the magnets normally, using the configuration database for settings:
173  setSISMagnet(true);
174  }
175  // measure the voltage offset errors:
176  coldCart_mp -> measureSISVoltageError();
177 
178  // set the magnets back to their prior state:
179  if (magnetPrior) {
180  setSISMagnet(true, 0, 1, &iSet01);
181  setSISMagnet(true, 0, 2, &iSet02);
182  setSISMagnet(true, 1, 1, &iSet11);
183  setSISMagnet(true, 1, 2, &iSet12);
184  } else
185  setSISMagnet(false);
186 
187  // resume monitoring:
188  pauseMonitor(false, false);
189  }

351 bool CartAssembly::lockPLL() {
352  if (!checkWCA("CartAssembly::lockPLL"))
353  return false;
354 
355  if (freqLO_m == 0.0)
356  return false;
357 
358  LOG(LM_INFO) << "Locking PLL...." << endl;
359 
360  double freqYIG = 0;
361  double freqREF = 0;
362  int sbLock = WCA_mp -> getPLLSidebandLockSelectSetting();
363  int coarseYIG = getCoarseYIG(freqYIG, freqREF, freqLO_m, freqFLOOG_m, sbLock);
364  if (coarseYIG < 0) {
365  coarseYIG = 0;
366  return false;
367  }
368 
369  // lock search window is +/- 50 MHz:
370  const double window(0.05);
371 
372  // lock search step size is 3 MHz:
373  const double step(0.003);
374 
375  // compute YIG step size:
376  double stepYIG = (config_m.WCA_m.FHIYIG_m - config_m.WCA_m.FLOYIG_m) / 4095.0;
377 
378  // lock search window in YIG steps:
379  int windowSteps = (int) (window / stepYIG) + 1;
380 
381  // lock step size in YIG steps:
382  int stepSize = (int) (step / stepYIG);
383  if (stepSize == 0)
384  stepSize = 1;
385 
386  pauseMonitor(true, true, "Searching for lock.");
387 
388  bool success = WCA_mp -> searchForLock(coarseYIG, windowSteps, stepSize);
389  if (success) {
390  // clear the unlock detect latch so we will see if lock is lost from here on:
391  WCA_mp -> clearUnlockDetect();
392  success = WCA_mp -> adjustPLL(0.0);
393  }
394  pauseMonitor(false, false);
395  return success;
396 }

940 // adjust the LO power amplifier drain voltage setting to achieve a target junction current.
941 // 'pol' controls which polarization to optimize
942 // 'VD' is the drain voltage to adjust.
943 // 'targetIJ1' is the optimal junction current for SIS1. SIS2 is allowed to fall where it may.
944 bool CartAssembly::adjustLOPASinglePol(int pol, float &VD, float targetIJ1, bool verboseLogging) {
945  if (!checkWCA("CartAssembly::adjustLOPASinglePol"))
946  return false;
947  if (!checkColdCart("CartAssembly::adjustLOPASinglePol"))
948  return false;
949 
950  if (!coldCart_mp -> getSISEnableSetting()) {
951  string msg("OptimizeLOPowerAmp ERROR: SIS mixer is not enabled.");
952  FEMCEventQueue::addStatusMessage(false, msg);
953  LOG(LM_ERROR) << msg << endl;
954  return false;
955  }
956  if (!WCA_mp -> getPAEnableSetting()) {
957  string msg("OptimizeLOPowerAmp ERROR: Power amplifier is not enabled.");
958  FEMCEventQueue::addStatusMessage(false, msg);
959  LOG(LM_ERROR) << msg << endl;
960  return false;
961  }
962 
963  // Constants governing the range of possible settings:
964  const float VDmax(2.5); // V
965  const float VDstep(VDmax / 255);
966  float window = fabsf(targetIJ1 * 0.05); // +/- 5% of target in uA
967  float startVD(VD);
968 
969  // Set a maximum iteration count to guard against pathological inputs:
970  const int iterLimit = 260;
971 
972  if (verboseLogging) {
973  LOG(LM_INFO) << "OptimizeLOPowerAmp: pol=" << pol << fixed << setprecision(4)
974  << " startVD=" << startVD << " VDstep=" << VDstep << " targetIJ1=" << targetIJ1 << " +/- " << window << " uA" << endl;
975  }
976 
977  // Check for targetIJ1==0 and just use starting VD if so:
978  if (targetIJ1 == 0.0) {
979  WCA_mp -> setPADrainVoltage(pol, VD);
980  LOG(LM_ERROR) << "OptimizeLOPowerAmp: targetIJ1 is zero so setting VD, can't optimize." << endl;
981  return true;
982  }
983 
984  // Define the window of IJ values near the target. We'll try to move IJ1 within these limits:
985  float loThreshold = fabsf(targetIJ1) - window;
986  if (loThreshold < 0)
987  loThreshold = 0;
988  float hiThreshold = fabsf(targetIJ1) + window;
989 
990  // Initialize variables used within the loop:
991  bool done = false;
992  bool error = false;
993  int iter = 0;
994  bool foundMin = false; // true if a minimum has been seen
995  float minErrorIJ1 = 999; // starting smallest errorIJ1 values seen
996  float minErrorVD1 = 0; // VD setting where smallest errorIJ1 was seen
997  float VJ1; // instantaneous VJ reading
998  float IJ1; // instantaneous IJ reading
999  float errorIJ1; // IJ error = abs(IJ1 - targetIJ1)
1000  int searchDirection; // inverse of the sign of the error
1001  vector<float> VDHistory; // array to collect recent VD settings to prevent oscillating around the target.
1002 
1003  while (!done && !error) {
1004 
1005  // Check for maximum iterations:
1006  if (iter++ > iterLimit) {
1007  string msg("OptimizeLOPowerAmp ERROR: Max iterations exceeded.");
1008  FEMCEventQueue::addStatusMessage(false, msg);
1009  LOG(LM_ERROR) << fixed << "VD=" << setprecision(4) << VD << " IJ1=" << IJ1 << " iter=" << iter << " Max iterations exceeded, error stop." << endl;
1010  error = true;
1011 
1012  // Check for VD out of range:
1013  } else if (VD < 0 || VD > VDmax) {
1014  string msg("OptimizeLOPowerAmp ERROR: VD out of range.");
1015  FEMCEventQueue::addStatusMessage(false, msg);
1016  LOG(LM_ERROR) << fixed << "VD=" << setprecision(4) << VD << " IJ1=" << IJ1 << " iter=" << iter << " VD out of range, error stop." << endl;
1017  error = true;
1018 
1019  } else {
1020  // Set the new (or starting) VD and get the instantaneous VJ & IJ reading:
1021  WCA_mp -> setPADrainVoltage(pol, VD);
1022  VJ1 = coldCart_mp -> getSISVoltage(pol, 1);
1023  IJ1 = 1000 * fabsf(coldCart_mp -> getSISCurrent(pol, 1, 10));
1024  // Save the drain voltage setting into the history array:
1025  VDHistory.push_back(VD);
1026  // Limit the history array to four elements:
1027  if (VDHistory.size() == 5)
1028  VDHistory.erase(VDHistory.begin());
1029  std::stringstream historyStream;
1030 
1031  if (verboseLogging) {
1032  historyStream << fixed << setprecision(4);
1033  for (int index = VDHistory.size() - 1; index >= 0; --index)
1034  historyStream << " " << VDHistory[index];
1035  }
1036 
1037  if (IJ1 >= loThreshold && IJ1 <= hiThreshold) {
1038  // We are within the window.
1039  if (verboseLogging)
1040  LOG(LM_INFO) << fixed << "VD=" << setprecision(4) << historyStream.str() << " IJ1=" << IJ1 << " Within window." << endl;
1041 
1042  // Compute the IJ error and sign:
1043  // We pretend zero error is positive to simplify.
1044  errorIJ1 = IJ1 - targetIJ1;
1045  if (errorIJ1 >= 0)
1046  searchDirection = -1;
1047  else {
1048  searchDirection = 1;
1049  errorIJ1 = -errorIJ1;
1050  }
1051 
1052  // Find the minimum error...
1053 
1054  // If the absolute error is smaller than the smallest seen,
1055  // save the new smallest and the corresponding VD:
1056  if (!foundMin) {
1057  if (errorIJ1 <= minErrorIJ1) {
1058  minErrorIJ1 = errorIJ1;
1059  minErrorVD1 = VD;
1060  // Otherwise the error has started to increase; stop here:
1061  } else {
1062  VD = minErrorVD1;
1063  if (verboseLogging)
1064  LOG(LM_INFO) << fixed << "VD=" << setprecision(4) << historyStream.str() << " errorIJ1=" << minErrorIJ1 << " Found minimum, stopping." << endl;
1065  done = foundMin = true;
1066  }
1067  }
1068  if (!foundMin)
1069  VD += searchDirection * VDstep;
1070 
1071  // Check for oscillations, defined as two iterations through the same two values:
1072  } else if (VDHistory.size() == 4 && VDHistory[0] == VDHistory[2] && VDHistory[1] == VDHistory[3]) {
1073  // take the larger of the two values:
1074  if (VDHistory[0] > VD) {
1075  VD = VDHistory[0];
1076  // Set it to the hardware:
1077  WCA_mp -> setPADrainVoltage(pol, VD);
1078  VJ1 = coldCart_mp -> getSISVoltage(pol, 1);
1079  IJ1 = 1000 * fabsf(coldCart_mp -> getSISCurrent(pol, 1, 10));
1080  }
1081  // log a message and exit:
1082  if (verboseLogging)
1083  LOG(LM_INFO) << fixed << "VD=" << setprecision(4) << historyStream.str() << " IJ1=" << IJ1 << " Oscillation detected, stopping." << endl;
1084  done = foundMin = true;
1085 
1086  // If IJ is below the low threshold, step up:
1087  } else if (IJ1 < loThreshold) {
1088  if (verboseLogging)
1089  LOG(LM_INFO) << fixed << "VD=" << setprecision(4) << historyStream.str() << " IJ1=" << IJ1 << " Stepping up." << endl;
1090  VD += VDstep;
1091 
1092  // If it is above the high threshold step down:
1093  } else if (IJ1 > hiThreshold) {
1094  if (verboseLogging)
1095  LOG(LM_INFO) << fixed << "VD=" << setprecision(4) << historyStream.str() << " IJ1=" << IJ1 << " Stepping down." << endl;
1096  VD -= VDstep;
1097  }
1098  }
1099  }
1100  // Set the final VD:
1101  if (error)
1102  VD = startVD;
1103  WCA_mp -> setPADrainVoltage(pol, VD);
1104  return foundMin;
1105 }
