143 FEHardwareDevice::FEMC_ERROR CryostatImplBase::CryostatMonitorTempsensor(AmbRelativeAddr RCA, float &target) {
144  FEHardwareDevice::FEMC_ERROR ret(FEMC_NO_ERROR);
145  sem_t synchLock;
146  sem_init(&synchLock, 0, 0);
147  ret = syncMonitor(RCA, target, synchLock);
148  // if AMB error, quit without retrying:
149  if (ret == FEMC_AMB_ERROR)
150  return ret;
151  // no AMB error. Check for retry required:
152  else if (ret == FEMC_HARDW_RETRY_WARN)
153  ret = FEMC_NO_ERROR; // Proceed to 2nd measurement.
154  SLEEP(25);
155  syncMonitor(RCA, target, synchLock);
156  sem_destroy(&synchLock);
157  return ret;
158 }

305 void CryostatImplBase::monitorAction(Time *timestamp_p) {
306 
307  if (!timestamp_p)
308  return;
309 
310  // constant factor to convert 100 ns ticks to ms:
311  static const Time milliseconds = 10000;
312  static const Time monitorInterval = 50 * milliseconds;
313 
314  bool doMonitor = false;
315 
316  if (lastMonitorTime == 0 && logMonitors_m)
317  logMon(true);
318 
319  if (lastMonitorTime == 0 || (*timestamp_p - lastMonitorTime >= monitorInterval)) {
320  lastMonitorTime = *timestamp_p;
321  doMonitor = true;
322  }
323 
324  if (doMonitor) {
325  switch (monitorPhase) {
326  case 0:
327  if (!executeNextMon()) {
328  if (randomizeAnalogMonitors_m)
329  randomizeMon();
330  monitorPhase = 1;
331  }
332  break;
333  case 1:
334  backingPumpEnable_value = backingPumpEnable();
335  ++monitorPhase;
336  break;
337  case 2:
338  turboPumpEnable_value = turboPumpEnable();
339  ++monitorPhase;
340  break;
341  case 3:
342  turboPumpErrorState_value = turboPumpErrorState();
343  ++monitorPhase;
344  break;
345  case 4:
346  turboPumpHighSpeed_value = turboPumpHighSpeed();
347  ++monitorPhase;
348  break;
349  case 5:
350  gateValveState_value = gateValveState();
351  ++monitorPhase;
352  break;
353  case 6:
354  solenoidValveState_value = solenoidValveState();
355  ++monitorPhase;
356  break;
357  case 7:
358  vacuumGaugeEnable_value = vacuumGaugeEnable();
359  ++monitorPhase;
360  break;
361  case 8:
362  vacuumGaugeErrorState_value = vacuumGaugeErrorState();
363  ++monitorPhase;
364  // no break;
365  default:
366  if (logMonitors_m)
367  logMon();
368  monitorPhase = 0;
369  break;
370  };
371  }
372 }
