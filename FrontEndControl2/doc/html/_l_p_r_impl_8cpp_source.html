36 bool LPRImpl::setObservingBand(int band) {
37  if (band < 1 || band > 10)
38  return false;
39  opticalSwitchPort((unsigned char) (band - 1));
40  observingBand_m = band;
41  unsigned long beginTime = GETTIMER();
42  unsigned long elapsed;
43  bool timeout = false;
44  while (!timeout && opticalSwitchBusy()) {
45  SLEEP(5);
46  elapsed = GETTIMER() - beginTime;
47  if (elapsed > 2000)
48  timeout = true;
49  }
50  return !timeout;
51 }
