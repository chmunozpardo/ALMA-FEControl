190 void IFSwitchImplBase::monitorAction(Time *timestamp_p) {
191 
192  if (!timestamp_p)
193  return;
194 
195  // constant factor to convert 100 ns ticks to ms:
196  static const Time milliseconds = 10000;
197  static const Time monitorInterval = 2000 * milliseconds;
198 
199  bool doMonitor = false;
200 
201  if (lastMonitorTime == 0 && logMonitors_m)
202  logMon(true);
203 
204  if (lastMonitorTime == 0 || (*timestamp_p - lastMonitorTime >= monitorInterval)) {
205  lastMonitorTime = *timestamp_p;
206  doMonitor = true;
207  }
208 
209  if (doMonitor) {
210  switchCartridge_value = switchCartridge();
211  pol0Sb1Attenuation_value = pol0Sb1Attenuation();
212  pol0Sb2Attenuation_value = pol0Sb2Attenuation();
213  pol1Sb1Attenuation_value = pol1Sb1Attenuation();
214  pol1Sb2Attenuation_value = pol1Sb2Attenuation();
215  pol0Sb1TempServoEnable_value = pol0Sb1TempServoEnable();
216  pol0Sb2TempServoEnable_value = pol0Sb2TempServoEnable();
217  pol1Sb1TempServoEnable_value = pol1Sb1TempServoEnable();
218  pol1Sb2TempServoEnable_value = pol1Sb2TempServoEnable();
219  while (executeNextMon()) ;
220  if (randomizeAnalogMonitors_m)
221  randomizeMon();
222  if (logMonitors_m)
223  logMon();
224  }
225 }
