186 bool WCAImpl::testAlternateLockDetect() {
187  if (!testPLLPowerLevels()) {
188  LOG(LM_DEBUG) << "WCAImpl::testAlternateLockDetect: failed testPLLPowerLevels" << endl;
189  return false;
190  }
191  short yto = getYTOCoarseTuneSetting();
192  float cv0 = pllCorrectionVoltage();
193  float cv1, cv2;
194  bool stepsOk(false), statsOk(false);
195  // move the YTO to +2 steps, then +1, measuring CV at each step:
196  if (testPLLCVInRange(cv0)) {
197  LOG(LM_INFO) << "WCAImpl::testAlternateLock: yto=" << yto << " cv0=" << cv0 << endl;
198  ytoCoarseTune(yto + 2);
199  SLEEP(1);
200  cv2 = pllCorrectionVoltage();
201  ytoCoarseTune(yto + 1);
202  SLEEP(1);
203  cv1 = pllCorrectionVoltage();
204  // Move back to starting YTO:
205  ytoCoarseTune(yto);
206  // When locked, the correction voltage decreases as the YTO tuning increases:
207  stepsOk = (cv0 > cv1 && cv1 > cv2);
208  }
209  if (stepsOk) {
210  // read several more times to collect stats on the variance of the CV:
211  SLEEP(1);
212  pllCorrectionVoltage();
213  SLEEP(1);
214  pllCorrectionVoltage();
215  SLEEP(1);
216  pllCorrectionVoltage();
217  SLEEP(1);
218  pllCorrectionVoltage();
219  SLEEP(1);
220  pllCorrectionVoltage();
221  statsOk = testPLLCorrectionVoltageStats();
222  }
223  if (stepsOk) {
224  LOG(LM_INFO) << "WCAImpl::testAlternateLock: cv1=" << cv1 << " cv2=" << cv2
225  << " success=" << (stepsOk && statsOk)
226  << " cvSize=" << pllCorrectionVoltage_count
227  << " cvMin=" << pllCorrectionVoltage_min
228  << " cvMax=" << pllCorrectionVoltage_max
229  << endl;
230  }
231  return stepsOk && statsOk;
232 }

244 bool WCAImpl::monitorLockForDisplay() const {
245  if (fabsf(pllRefTotalPower_value) < lowRefTotalPower)
246  return false; // the ref total power is now too low.
247  if (fabsf(pllIfTotalPower_value) < lowIFTotalPower)
248  return false; // the IF total power is now too low.
249 
250  if (!alternateLockTest_m) {
251  // we were previously locked in the 'normal' way.
252  if (pllLockDetectVoltage_value < lowLockDetectVoltage)
253  return false; // ...but lock detect voltage has dropped below threshold.
254  } else {
255  // previous lock was in the 'alternate' way.
256  if (pllNullLoopIntegrator_value)
257  return false; // ...but now the PLL is defeated so it can't be considered locked.
258  if (!testPLLCVInRange(pllCorrectionVoltage_value))
259  return false; // ...but now the PLL cv is out of range.
260  if (!testPLLCorrectionVoltageStats()) {
261  LOG(LM_DEBUG) << "WCAImpl::monitorLockForDisplay: no lock"
262  << " cvSize=" << pllCorrectionVoltage_count
263  << " cvMin=" << pllCorrectionVoltage_min
264  << " cvMax=" << pllCorrectionVoltage_max << endl;
265  return false; // ...but now the correction voltage variance is too large.
266  }
267  }
268  // appears we are still locked:
269  return true;
270 }

275 bool WCAImpl::searchForLock(int &coarseYIG, int windowSteps, int stepSize) {
276  int loLimit = coarseYIG - windowSteps;
277  if (loLimit < 0)
278  loLimit = 0;
279  int hiLimit = coarseYIG + windowSteps;
280  if (hiLimit > 4095)
281  hiLimit = 4095;
282  int tryCoarseYIG = coarseYIG;
283  int stepSleep = 2; // ms - YTO moves to within 10% of commanded value after 1 ms.
284  bool lock = false;
285  bool error = false;
286 
287  isLockedLO_m = alternateLockTest_m = false;
288 
289  // outer loop tries normal lock search and then alternateLockTest for bands 9 & 10:
290  while (!lock && !error) {
291  ytoCoarseTuneWithTrace(coarseYIG, "WCAImpl::searchForLock initial");
292  // give the YTO time to slew:
293  SLEEP(stepSleep);
294  LOG(LM_INFO) << "Searching for Lock" << ((alternateLockTest_m) ? " with alternateLockTest..." : "...") << endl;
295 
296  // initialize stepping size and direction:
297  int step = stepSize;
298 
299  // inner loop searches for lock using standard or alternateLockTest strategy:
300  while (!lock && !error) {
301  if (!alternateLockTest_m)
302  lock = testPLLLockDetectVoltage();
303  else
304  lock = testAlternateLockDetect();
305 
306  if (!lock) {
307  // no lock so move the YTO to the next try frequency:
308  tryCoarseYIG = coarseYIG + step;
309  if (tryCoarseYIG >= loLimit && tryCoarseYIG <= hiLimit) {
310  pllNullLoopIntegrator(true);
311  ytoCoarseTuneWithTrace(tryCoarseYIG, "WCAImpl::searchForLock");
312  SLEEP(stepSleep);
313  pllNullLoopIntegrator(false);
314  clearUnlockDetect();
315  }
316  // step increases by stepsize or alternates sign each iteration
317  // to expand the search outward from the starting value:
318  if (step > 0)
319  step = -step;
320  else {
321  step = -step + stepSize;
322  if (step > windowSteps)
323  error = true;
324  }
325  }
326  }
327  // can we try again with the alternateLockTest strategy?
328  if (!lock && !alternateLockTest_m && tryAlternateLockMethod && bandSupportsAlternateLockMethod()) {
329  // retry inner loop again:
330  alternateLockTest_m = true;
331  error = false;
332  }
333  }
334  if (lock)
335  coarseYIG = tryCoarseYIG;
336  else
337  alternateLockTest_m = false;
338  isLockedLO_m = lock;
339  return isLockedLO_m;
340 }

342 bool WCAImpl::adjustPLL(float targetCorrVoltage) {
343  LOG(LM_INFO) << "WCAImpl::adjustPLL targetCV=" << targetCorrVoltage << endl;
344 
345  int maxDist(50);
346  int dist(0);
347  double window(0.25);
348  double hiThreshold = targetCorrVoltage + window;
349  double loThreshold = targetCorrVoltage - window;
350 
351  float correctionVoltage = pllCorrectionVoltage();
352  float voltageError = correctionVoltage - targetCorrVoltage;
353 
354  unsigned short coarseYIG = ytoCoarseTune();
355  int tryCoarseYIG = coarseYIG;
356 
357  LOG(LM_INFO) << "WCAImpl::adjustPLL CV=" << correctionVoltage << " vError=" << voltageError << " coarseYIG=" << tryCoarseYIG << endl;
358 
359  if (tryCoarseYIG >= 0)
360  ytoCoarseTuneWithTrace(tryCoarseYIG, "WCAImpl::adjustPLL initial");
361  else
362  return false;
363 
364  bool done = false;
365  bool error = false;
366  int retries = 50;
367  while (!done && !error) {
368  correctionVoltage = pllCorrectionVoltage();
369  if (correctionVoltage >= loThreshold && correctionVoltage <= hiThreshold)
370  done = true;
371  else if (--retries <= 0) {
372  done = true;
373  LOG(LM_INFO) << "WCAImpl::adjustPLL too many retries." << endl;
374  } else {
375  voltageError = correctionVoltage - targetCorrVoltage;
376  tryCoarseYIG += (voltageError > 0) ? 1 : -1;
377  dist = tryCoarseYIG - coarseYIG;
378  if (dist < 0)
379  dist = -dist;
380  if (dist > maxDist)
381  error = true;
382  else if (tryCoarseYIG >= 0 && tryCoarseYIG <= 4095)
383  ytoCoarseTuneWithTrace(tryCoarseYIG, "WCAImpl::adjustPLL");
384  }
385  }
386 
387  if (!interrogateLock()) {
388  string msg("WCAImpl ERROR: band ");
389  msg += to_string(band_m);
390  msg += " lost the lock while adjusting the PLL.";
391  FEMCEventQueue::addStatusMessage(false, msg);
392  LOG(LM_ERROR) << msg << endl;
393  error = true;
394  } else
395  // clear the unlock detect latch so we will see if lock is lost hereafter:
396  clearUnlockDetect();
397  return done && !error;
398 }

464 void WCAImpl::setEnableLO(bool enable, bool setAMC, bool detune) {
465  paEnable_m = LOEnable_m = enable;
466  if (!enable) {
467  // turn off the LO PA:
468  WCAImplBase::paPol0DrainVoltage(0.0);
469  WCAImplBase::paPol1DrainVoltage(0.0);
470  // Set the null integrator so we can return to lock on enable:
471  WCAImplBase::pllNullLoopIntegrator(true);
472  if (detune) {
473  // since there still might be leakage, detune the YTO to the far end of the band:
474  WCAImplBase::ytoCoarseTune((unsigned short) (ytoCoarseTune_m < 2048) ? 4095 : 0);
475  }
476  if (setAMC && band_m == 3) {
477  // turn off the AMC stage E:
478  WCAImpl::setAMC(0.0, amcGateEVoltage_m);
479  }
480  } else {
481  // set the AMC back where it was:
482  if (setAMC && band_m == 3) {
483  // turn back on the AMC stage E with the prior settings:
484  WCAImpl::setAMC(amcDrainEVoltage_m, amcGateEVoltage_m);
485  }
486  // Set the YTO back where it was:
487  WCAImplBase::ytoCoarseTune(ytoCoarseTune_m);
488  // Reset the null integrator to its previous state. Should re-lock if was locked before:
489  WCAImplBase::pllNullLoopIntegrator(pllNullLoopIntegrator_m);
490  // Turn on the LO PA:
491  WCAImplBase::paPol0DrainVoltage(paDrainVoltage_m[0]);
492  WCAImplBase::paPol1DrainVoltage(paDrainVoltage_m[1]);
493  // Clear the unlock detect latch:
494  clearUnlockDetect();
495  }
