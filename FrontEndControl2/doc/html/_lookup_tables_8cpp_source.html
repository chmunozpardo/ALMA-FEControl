59 bool ParamTable::get(const double &freq, ParamTableRow &target, bool interpolate) const {
60  target.clear();
61  // check for nothing to return:
62  if (base_type::empty())
63  return false;
64 
65  // find the first item whose key is not less than freq:
66  const_iterator it(base_type::lower_bound(freq));
67  double foundFreq(0.0);
68 
69  // check for past the last element:
70  if (it == base_type::end()) {
71  // back up one and return the last element:
72  --it;
73  foundFreq = it -> first;
74  target = it -> second;
75  return true;
76  }
77  // check for freq <= the first element
78  if (it == base_type::begin()) {
79  // return the first element:
80  foundFreq = it -> first;
81  target = it -> second;
82  return true;
83  }
84  // check for exact frequency match:
85  double freqHi = it -> first;
86  if (freqHi == freq) {
87  foundFreq = it -> first;
88  target = it -> second;
89  return true;
90  }
91  // back up by one and calculate the weighting between the lower and upper elements:
92  const_iterator it2 = it;
93  --it;
94  double freqLo = it -> first;
95  double weight = calculateWeight(freq, freqLo, freqHi);
96 
97  // check for interval smaller than minInterpolateInterval_m:
98  if ((freqHi - freqLo) <= minInterpolateInterval_m)
99  interpolate = false;
100 
101  // if not interpolating, return one or the other:
102  if (!interpolate) {
103  if (weight >= 0.5) {
104  // return the upper one:
105  foundFreq = freqHi;
106  target = it2 -> second;
107  } else {
108  // return the lower one:
109  foundFreq = freqLo;
110  target = it -> second;
111  }
112 
113  // interpolate:
114  } else {
115  // don't assign foundFreq;
116  target = it -> second;
117  target.weightedAdd(weight, it2 -> second);
118  }
119  return true;
120 }
