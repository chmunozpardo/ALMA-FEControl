54 DLLEXPORT short LORTMControlInit() {
55  ++connectedLORTMClients;
56  if (LORTM && LORTMValid) {
57  LOG(LM_INFO) << "FEControlInit: connectedLORTMClients=" << connectedLORTMClients << endl;
58  return 0;
59  }
60  if (LVWrapperInit() < 0) {
61  LOG(LM_ERROR) << "FEControlInit: LVWrapperInit failed." << endl;
62  return -1;
63  }
64  LOG(LM_INFO) << "FEControlInit: connectedLORTMClients=" << connectedLORTMClients << endl;
65  try {
66  // get the path where the FrontendControlDLL.ini file is located:
67  string iniPath, tmp;
68  splitPath(iniFileName, iniPath, tmp);
69  if (iniPath.empty())
70  iniPath = ".";
71 
72  // open the FrontendControlDLL.ini file:
73  CIniFile configINI(iniFileName);
74  configINI.ReadFile();
75 
76  // look for the item specifying a separate file for signal source config:
77  tmp = configINI.GetValue("configFiles", "TeraxionLORTM");
78  if (tmp.empty())
79  // not found. use the FrontendControlDLL.ini file to load signal source config:
80  LORTMIniFilename = iniFileName;
81  else {
82  // if found, open that file to load the signal source config data from:
83  LORTMIniFilename = iniPath + "/" + tmp;
84  LOG(LM_INFO) << "Using Teraxion LORTM configuration file '" << LORTMIniFilename << "'" << endl;
85  configINI.Clear();
86  configINI.SetPath(LORTMIniFilename);
87  if (!configINI.ReadFile()) {
88  LOG(LM_ERROR) << "Error reading Teraxion LORTM config file." << endl;
89  return -1;
90  }
91  }
92 
93  tmp = configINI.GetValue("connection", "channelLORTM");
94  if (!tmp.empty())
95  CANChannelLORTM = from_string<unsigned long>(tmp);
96  tmp = configINI.GetValue("connection", "nodeAddressLORTM");
97  if (!tmp.empty()) {
98  if (tmp[0] == '0' && (tmp[1] == 'x' || tmp[1] == 'X'))
99  tmp = tmp.substr(2);
100  nodeAddressLORTM = from_string<unsigned long>(tmp, std::hex);
101  }
102  if (nodeAddressLORTM > 0) {
103  LOG(LM_INFO) << "Connecting to LORTM on CAN" << CANChannelLORTM
104  << " nodeAddress=" << uppercase << hex << setw(2) << setfill('0') << nodeAddressLORTM << dec << setw(0) << endl;
105  }
106  tmp = configINI.GetValue("LORTM", "freqRefLaser");
107  if (!tmp.empty()) {
108  freqRefLaser = from_string<double>(tmp);
109  LOG(LM_INFO) << "LORTM configuration: freqRefLaser=" << fixed << setprecision(6) << freqRefLaser << " GHz" << endl;
110  }
111 
112  } catch (...) {
113  LOG(LM_ERROR) << "LORTMControlInit exception loading configuration file.." << endl;
114  return -1;
115  }
116 
117  if (nodeAddressLORTM)
118  LORTM = new LORTMImpl(CANChannelLORTM, nodeAddressLORTM);
119  else
120  LOG(LM_ERROR) << "LORTMControlInit failed. nodeAddressLORTM is 0." << endl;
121 
122  if (LORTM) {
123  LORTMValid = true;
124  if (freqRefLaser > 0.0)
125  LORTM -> setFreqRefLaser(freqRefLaser);
126  LORTM -> startMonitor();
127  return 0;
128  } else
129  return -1;
130 }

132 DLLEXPORT short LORTMControlShutdown() {
133  if (!LORTMValid)
134  return -1;
135  --connectedLORTMClients;
136  LOG(LM_INFO) << "LORTMControlShutdown: connectedLORTMClients=" << connectedLORTMClients << endl;
137  if (connectedLORTMClients <= 0) {
138  LORTMValid = false;
139  LOG(LM_INFO) << "LORTMControlShutdown..." << endl;
140  if (LORTM) {
141  LORTM -> stopMonitor();
142  LORTM -> shutdown();
143  WHACK(LORTM);
144  LOG(LM_INFO) << "LORTMControlShutdown: LORTM destroyed." << endl;
145  }
146  LVWrapperShutdown();
147  }
148  return 0;
149 }
