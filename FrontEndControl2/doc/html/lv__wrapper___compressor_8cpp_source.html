46 DLLEXPORT short CompressorControlInit() {
47  ++connectedCompressorClients;
48 
49  if (Compressor && CompressorValid) {
50  LOG(LM_INFO) << "CompressorControlInit: connectedCompressorClients=" << connectedCompressorClients << endl;
51  return 0;
52  }
53  if (LVWrapperInit() < 0) {
54  LOG(LM_ERROR) << "CompressorControlInit: LVWrapperInit failed." << endl;
55  return -1;
56  }
57  LOG(LM_INFO) << "CompressorControlInit: connectedCompressorClients=" << connectedCompressorClients << endl;
58  try {
59  CIniFile configINI(iniFileName);
60  configINI.ReadFile();
61  std::string tmp;
62 
63  tmp = configINI.GetValue("connection", "channelCompressor");
64  if (!tmp.empty())
65  CANChannelCompressor = from_string<unsigned long>(tmp);
66  tmp = configINI.GetValue("connection", "nodeAddressCompressor");
67  if (!tmp.empty()) {
68  if (tmp[0] == '0' && (tmp[1] == 'x' || tmp[1] == 'X'))
69  tmp = tmp.substr(2);
70  nodeAddressCompressor = from_string<unsigned long>(tmp, std::hex);
71  }
72  if (nodeAddressCompressor > 0) {
73  LOG(LM_INFO) << "Connecting to Compressor M&C module on CAN" << CANChannelCompressor
74  << " nodeAddress=" << uppercase << hex << setw(2) << setfill('0') << nodeAddressCompressor << dec << setw(0) << endl;
75  }
76 
77  } catch (...) {
78  LOG(LM_ERROR) << "CompressorControlInit exception loading configuration file.." << endl;
79  return -1;
80  }
81 
82  if (nodeAddressCompressor)
83  Compressor = new CompressorImpl(CANChannelCompressor, nodeAddressCompressor);
84  else
85  LOG(LM_ERROR) << "CompressorControlInit failed. nodeAddressCompressor is 0." << endl;
86 
87  if (Compressor) {
88  CompressorValid = true;
89  Compressor -> setThermalLogInterval(thermalLogInterval);
90  Compressor -> startMonitor();
91  return 0;
92  } else
93  return -1;
94 }

96 DLLEXPORT short CompressorControlShutdown() {
97  if (!CompressorValid)
98  return -1;
99  --connectedCompressorClients;
100  LOG(LM_INFO) << "CompressorControlShutdown: connectedCompressorClients=" << connectedCompressorClients << endl;
101  if (connectedCompressorClients <= 0) {
102  CompressorValid = false;
103  LOG(LM_INFO) << "CompressorControlShutdown..." << endl;
104  if (Compressor) {
105  Compressor -> stopMonitor();
106  Compressor -> shutdown();
107  WHACK(Compressor);
108  LOG(LM_INFO) << "CompressorControlShutdown: Compressor destroyed." << endl;
109  }
110  LVWrapperShutdown();
111  }
112  return 0;
113 }
