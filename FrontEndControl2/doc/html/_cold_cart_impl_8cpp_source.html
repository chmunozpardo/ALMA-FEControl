153  // ignore if no SIS:
154  if (!hasSIS())
155  return;
156 
157  // ignore sweep parameter if this is band 6 or below:
158  if (band_m <= 6)
159  sweep = false;
160 
161  // cache the prior value as valNow:
162  float valNow = getSISVoltageSetting(pol, sb);
163 
164  // declare a pointer to void member function with float parameter:
165  void (ColdCartImpl::* pf) (float) = NULL;
166 
167  // assign the appropriate member function to the pointer:
168  if (pol == 0) {
169  if (sb == 1)
170  pf = &ColdCartImpl::sisPol0Sb1Voltage;
171  else if (sb == 2)
172  pf = &ColdCartImpl::sisPol0Sb2Voltage;
173  } else if (pol == 1) {
174  if (sb == 1)
175  pf = &ColdCartImpl::sisPol1Sb1Voltage;
176  else if (sb == 2)
177  pf = &ColdCartImpl::sisPol1Sb2Voltage;
178  }
179 
180  // decide whether to sweep from the current value to the new value:
181  float span = val - valNow;
182  if (sweep && span != 0.0) {
183  float neg = (span < 0);
184  float step = (neg) ? -0.05 : 0.05;

186  bool done = false;
187  while (!done) {
188  if ((neg && valNow <= val) || (!neg && valNow >= val))
189  done = true;
190  else {
191  // set the junction voltage using the function pointer:
192  (this ->* pf)(valNow);
193  valNow += step;
194  }
195  }
196  }
197  // set the final value using the function pointer:
198  (this ->* pf)(val);

216  if (!hasSIS())
217  return 0.0;

220  if (sb == 2 && !hasSb2())
221  return 0.0;

242  testVoltage = 2.2;
243  break;

245  testVoltage = 2.3;
246  break;
247  case 10:
248  testVoltage = 2.1;

278 bool ColdCartImpl::measureIVCurve(XYPlotArray &target, int pol, int sb, float VJlow, float VJhigh, float VJstep) {
279  if (!hasSIS()) {
280  string msg("measureIVCurve ERROR: can't measure SIS for band ");
281  msg += to_string(band_m);
282  FEMCEventQueue::addStatusMessage(false, msg);
283  LOG(LM_ERROR) << "ColdCartImpl::" << msg << endl;
284  return false;
285  }
286  if (!getSISEnableSetting()) {
287  string msg("measureIVCurve ERROR: the SIS is not enabled.");
288  FEMCEventQueue::addStatusMessage(false, msg);
289  LOG(LM_ERROR) << "ColdCartImpl::" << msg << endl;
290  return false;
291  }
292 
293  // clear the stop flag:
294  ivCurveStop_m = false;
295 
296  // clear the output data array:
297  target.clear();
298 
299  // store the voltage setting in effect now:
300  float VJnom = getSISVoltageSetting(pol, sb);
301 
302  float VJset = 0;
303  float VJ = 0;
304  float IJ = 0;
305  float span = VJhigh - VJlow;
306 
307  if (span <= 0) {
308  string msg("measureIVCurve ERROR: VJHigh must be greater than VJLow.");
309  FEMCEventQueue::addStatusMessage(false, msg);
310  LOG(LM_ERROR) << "ColdCartImpl::" << msg << endl;
311  return false;
312  }
313 
314  if (VJstep < 0 || VJstep > span) {
315  string msg("measureIVCurve ERROR: VJ step size is out of range.");
316  FEMCEventQueue::addStatusMessage(false, msg);
317  LOG(LM_ERROR) << "ColdCartImpl::" << msg << endl;
318  return false;
319  }
320 
321  int steps = (int) (span / VJstep) + 1;
322 
323  target.reserve(2 * steps + 1);
324 
325  short progress = 0;
326  short lastProgress = 0;
327  const int stepSleep = 1; // ms
328  const int voltAveraging = 3;
329  const int currAveraging = 3;
330 
331  VJset = VJlow;
332  // move slowly to the first point:
333  setSISVoltage(pol, sb, VJset, true);
334  SLEEP(10);
335 
336  for (int index = 0; index < steps && !ivCurveStop_m; ++index) {
337  VJset = VJlow + (index * VJstep);
338  // move quickly from point to point:
339  setSISVoltage(pol, sb, VJset, false);
340  SLEEP(stepSleep);
341  // read back the instantaneous voltage and current:
342  VJ = getSISVoltage(pol, sb, voltAveraging);
343  IJ = getSISCurrent(pol, sb, currAveraging) * 1000.0; // convert mA to uA
344  target.push_back(XYPlotPoint(VJset, VJ, IJ));
345  progress = index * 5 / steps;
346  if (progress > lastProgress) {
347  lastProgress = progress;
348  FEMCEventQueue::addProgressEvent(10 * progress);
349  }
350  }
351  FEMCEventQueue::addProgressEvent(50);
352  lastProgress = 0;
353  for (int index = 0; index < steps && !ivCurveStop_m; ++index) {
354  VJset = VJlow + ((steps - index - 1) * VJstep);
355  // move quickly from point to point:
356  setSISVoltage(pol, sb, VJset, false);
357  SLEEP(stepSleep);
358  // read back the instantaneous voltage and current:
359  VJ = getSISVoltage(pol, sb, voltAveraging);
360  IJ = getSISCurrent(pol, sb, currAveraging) * 1000.0; // convert mA to uA
361  target.push_back(XYPlotPoint(VJset, VJ, IJ));
362  progress = index * 5 / steps;
363  if (progress > lastProgress) {
364  lastProgress = progress;
365  FEMCEventQueue::addProgressEvent(10 * progress + 50);
366  }
367  }
368  // sweep the SIS voltage back to where it was:
369  setSISVoltage(pol, sb, VJnom, true);
370 
371  FEMCEventQueue::addProgressEvent(100);
372  return (!ivCurveStop_m);
373 }

651 bool ColdCartImpl::sisMixerHeatingProcess(int pol, float targetTemp, int timeout) {
652  resetMixerHeating();
653 
654  if (!hasSISHeater()) {
655  string msg("MixerHeating: no mixer heaters in band ");
656  msg += to_string(band_m);
657  FEMCEventQueue::addStatusMessage(false, msg);
658  LOG(LM_ERROR) << msg << endl;
659  return false;
660  }
661  if (!(pol == -1 || pol == 0 || pol == 1)) {
662  string msg("MixerHeating: pol must be 0 or 1 or -1 to indicate both.");
663  FEMCEventQueue::addStatusMessage(false, msg);
664  LOG(LM_ERROR) << msg << endl;
665  return false;
666  }
667  if (targetTemp < 0.0 || targetTemp > 20.0) {
668  // checking against 0.0 here instead of mixerHeatingMinTargetTemp_m to support option
669  // of always heating until the timeout occurs.
670  string msg("MixerHeating: targetTemp must be between 0 and 20 K.");
671  FEMCEventQueue::addStatusMessage(false, msg);
672  LOG(LM_ERROR) << msg << endl;
673  return false;
674  }
675  if (timeout < 0 || timeout > mixerHeatingMaxTimeout_m) {
676  string msg("MixerHeating: timeout must be between 0 and 120 seconds.");
677  FEMCEventQueue::addStatusMessage(false, msg);
678  LOG(LM_ERROR) << msg << endl;
679  return false;
680  }
681  if (band_m == 9 && timeout > mixerHeatingMaxTimeoutBand9_m) {
682  timeout = mixerHeatingMaxTimeoutBand9_m;
683  string msg("MixerHeating: for band 9 timeout is limited to ");
684  msg += to_string<int>(mixerHeatingMaxTimeoutBand9_m);
685  msg += " seconds.";
686  FEMCEventQueue::addStatusMessage(true, msg);
687  LOG(LM_INFO) << msg << endl;
688  }
689 
690  LOG(LM_INFO) << "MixerHeating start: band=" << band_m << " pol=" << pol << " targetTemp=" << targetTemp << " timeout=" << timeout << endl;
691 
692  const string &logDir = FEConfig::getLogDir();
693  if (!logDir.empty()) {
694  string fileName, bandTxt, polText;
695  Time ts;
696  setTimeStamp(&ts);
697  timestampToText(&ts, fileName, true);
698  getBandText(bandTxt);
699  polText = to_string(pol);
700  if (pol == -1)
701  polText = "Both";
702  fileName = logDir + "MixerHeating-" + bandTxt + "-pol" + polText + "-" + fileName + ".txt";
703  mixerHeatingdataFile_mp = new ofstream(fileName.c_str(), ios_base::trunc);
704  }
705  string msg("MixerHeating: process started.");
706  FEMCEventQueue::addStatusMessage(true, msg);
707 
708  sisMixerHeatingReadData(true); // write column headers to output file.
709 
710  // make sure the SIS magnets are off:
711  bool magnetsPriorState = getSISMagnetEnableSetting();
712  if (magnetsPriorState) {
713  string msg("MixerHeating ALERT: Turning off SIS magnets. They will not automatically be re-enabled.");
714  FEMCEventQueue::addStatusMessage(true, msg);
715  LOG(LM_INFO) << msg << endl;
716  }
717  setSISMagnetEnable(false); // this sweeps the magnet currents to zero.
718 
719  // measure the current and mixer temperatures with the heater switched off:
720  setSISHeaterEnable(0, false);
721  setSISHeaterEnable(1, false);
722  float offCurrentPol0(0.0), offCurrentPol1(0.0);
723  float offTemp0(0.0), offTemp1(0.0);
724  for (int i = 0; i < 10; ++i) {
725  offCurrentPol0 += getSISHeaterCurrent(0);
726  offCurrentPol1 += getSISHeaterCurrent(1);
727  offTemp0 += cartridgeTemperature2();
728  offTemp1 += cartridgeTemperature5();
729  SLEEP(50);
730  }
731  offCurrentPol0 /= 10.0;
732  offCurrentPol1 /= 10.0;
733  offTemp0 /= 10.0;
734  offTemp1 /= 10.0;
735 
736  LOG(LM_INFO) << fixed << setprecision(2) << "MixerHeating: base temp pol0=" << offTemp0 << " pol1=" << offTemp1
737  << " base current pol0=" << offCurrentPol0 << " pol1=" << offCurrentPol1 << endl;
738 
739  float thresholdPol0(offCurrentPol0 + 1.0); // threshold is off current + 1mA
740  float thresholdPol1(offCurrentPol1 + 1.0);
741  float thresholdOffTemp0(offTemp0 + 0.20); // threshold is base temp + 0.20 K.
742  float thresholdOffTemp1(offTemp1 + 0.20);
743 
744  // Initialize loop variables:
745  float heaterCurrentPol0(0), heaterCurrentPol1(0);
746  float mixerTempPol0(0), mixerTempPol1(0);
747  bool targetReachedPol0(false), targetReachedPol1(false);
748  bool heaterOnPol0(false), heaterOnPol1(false);
749  bool timedOut(false);
750  bool retrigger(false);
751  bool done(false);
752 
753  // enable the heaters:
754  // pol == -1 means operate both heaters simultaneously; else pol is 0 or 1:
755  if (pol == -1 || pol == 0)
756  setSISHeaterEnable(0, true); // set pol0 heater control line ON
757  if (pol == -1 || pol == 1)
758  setSISHeaterEnable(1, true); // set pol1 heater control line ON
759 
760  timeout *= 1000;
761  mixerHeatingtimerStart_m = GETTIMER(); // get the timer value in ms.
762  sisMixerHeatingReadData(); // write mixer temps and heater currents to output file.
763 
764  while (!done && !mixerHeatingStop_m) {
765  if (GETTIMER() - mixerHeatingtimerStart_m > timeout)
766  timedOut = done = true;
767  else {
768  SLEEP(20);
769 
770  sisMixerHeatingReadData(); // write mixer temps and heater currents to output file.
771 
772  // read the mixer temperatures:
773  mixerTempPol0 = cartridgeTemperature2();
774  mixerTempPol1 = cartridgeTemperature5();
775 
776  // check mixer temps against the target temperature:
777  if (targetTemp >= mixerHeatingMinTargetTemp_m) {
778  targetReachedPol0 = (mixerTempPol0 >= targetTemp);
779  targetReachedPol1 = (mixerTempPol1 >= targetTemp);
780  } else
781  // we will heat until the timeout occurs:
782  targetReachedPol0 = targetReachedPol1 = false;
783 
784  // read the heater currents:
785  heaterCurrentPol0 = getSISHeaterCurrent(0);
786  heaterCurrentPol1 = getSISHeaterCurrent(1);
787 
788  // check heater currents against threshold:
789  heaterOnPol0 = (heaterCurrentPol0 > thresholdPol0);
790  heaterOnPol1 = (heaterCurrentPol1 > thresholdPol1);
791 
792  // adjust loop exit conditions based on whether heating both pols or just one:
793  if (pol == 0)
794  targetReachedPol1 = true;
795  else if (pol == 1)
796  targetReachedPol0 = true;
797 
798  //LOG(LM_INFO) << "MixerHeating: heaterCurrentPol0=" << heaterCurrentPol0 << " heaterCurrentPol1=" << heaterCurrentPol1
799  // << " heaterOnPol0=" << heaterOnPol0 << " heaterOnPol1=" << heaterOnPol1
800  // << " mixerTempPol0=" << mixerTempPol0 << " mixerTempPol1=" << mixerTempPol1
801  // << " targetReachedPol0=" << targetReachedPol0 << " targetReachedPol1=" << targetReachedPol1 << endl;
802 
803  // if target temp reached, exit now:
804  if (targetReachedPol0 && targetReachedPol1)
805  done = true;
806 
807  // check for whether heaters need to be re-triggered:
808  else {
809  retrigger = true;
810  if (pol == 0 || pol == -1) {
811  if (heaterOnPol0)
812  retrigger = false;
813  }
814  if (pol == 1 || pol == -1) {
815  if (heaterOnPol1)
816  retrigger = false;
817  }
818  if (retrigger) {
819  //LOG(LM_INFO) << "MixerHeating: heaters have switched off." << endl;
820  // re-enable the heaters:
821  if (pol == -1 || pol == 0)
822  setSISHeaterEnable(0, false);
823  if (pol == -1 || pol == 1)
824  setSISHeaterEnable(1, false);
825  if (pol == -1 || pol == 0)
826  setSISHeaterEnable(0, true); // set pol0 heater control line ON
827  if (pol == -1 || pol == 1)
828  setSISHeaterEnable(1, true); // set pol1 heater control line ON
829  }
830  }
831  }
832  }
833 
834  if (timedOut && targetTemp >= mixerHeatingMinTargetTemp_m)
835  msg = "MixerHeating: timed out waiting for mixer temperature.";
836  else
837  msg = "MixerHeating: finished successfully.";
838  msg += " Waiting for temps to return to normal...";
839  LOG(LM_INFO) << msg << endl;
840  FEMCEventQueue::addStatusMessage(true, msg);
841 
842  // switch off the heaters:
843  setSISHeaterEnable(0, false);
844  setSISHeaterEnable(1, false);
845 
846  // wait for the temperatures to return below the threshold:
847  unsigned long mixerCooldownTimerStart = GETTIMER();
848  const unsigned long mixerCooldownTimeout = 5 * 60 * 1000; // 5 minutes.
849  timedOut = done = false;
850  while (!done && !mixerHeatingStop_m) {
851  sisMixerHeatingReadData(); // write mixer temps and heater currents to output file.
852  if (GETTIMER() - mixerCooldownTimerStart > mixerCooldownTimeout)
853  timedOut = done = true;
854  else {
855  SLEEP(1000);
856  mixerTempPol0 = cartridgeTemperature2();
857  mixerTempPol1 = cartridgeTemperature5();
858  if ((mixerTempPol0 < thresholdOffTemp0) && (mixerTempPol1 < thresholdOffTemp1))
859  done = true;
860  }
861  }
862 
863  if (timedOut)
864  msg = "MixerHeating: timed out waiting for mixer temps to return to normal.";
865  else
866  msg = "MixerHeating: finished successfully.";
867  LOG(LM_INFO) << msg << endl;
868  FEMCEventQueue::addStatusMessage(true, msg);
869 
870  // re-enable the SIS magnets if they were enabled before:
871  // This doesn't energize the magnets, just makes them enabled so that they may be set.
872  setSISMagnetEnable(magnetsPriorState);
873 
874  // flush the log file:
875  if (mixerHeatingdataFile_mp) {
876  (*mixerHeatingdataFile_mp).flush();
877  delete mixerHeatingdataFile_mp;
878  mixerHeatingdataFile_mp = NULL;
879  }
880 
881  // Send the measurement finished event:
882  if (mixerHeatingStop_m)
883  msg = "MixerHeating: stopped.";
884 
885  return true;
886 }
