64 bool ConfigProviderMySQL::getFrontEndConfig(unsigned keyFacility, unsigned keyFrontEnd, FrontEndConfig &target) const {
65  target.reset(keyFacility, keyFrontEnd);
66 
67  if (!dbObject_mp -> isConnected())
68  return false;
69  if (!keyFacility || !keyFrontEnd)
70  return false;
71 
72 
73  // lookup the SN and ESN from the Front_Ends table:
74  FEICDataBase::ID_T frontEndId(keyFacility, keyFrontEnd);
75 
76  if (!dbObject_mp -> getFrontEndRecord(frontEndId, &target.SN_m, &target.ESN_m))
77  return false;
78 
79  // lookup the latest configuration ID from the FE_Config table:
80  FEICDataBase::ID_T configId = dbObject_mp -> getConfigId(frontEndId);
81 
82  FEICDataBase::ID_T componentId;
83  string SN;
84 
85  if (dbObject_mp -> getConfigCryostat(configId, componentId, SN)) {
86  // get the cryostat configuration:
87  CryostatConfig cryoConfig;
88  if (getCryostatConfig(componentId.keyFacility, componentId.keyId, cryoConfig))
89  target.setCryostatConfig(cryoConfig);
90  }
91 
92  if (dbObject_mp -> getConfigLPR(configId, componentId, SN)) {
93  // get the LPR configuration:
94  LPRConfig lprConfig;
95  if (getLPRConfig(componentId.keyFacility, componentId.keyId, lprConfig))
96  target.setLPRConfig(lprConfig);
97  }
98 
99  // get the FETIM component ID:
100  if (dbObject_mp -> getConfigFETIM(configId, componentId, SN)) {
101  target.fkFETIM_m = componentId.keyId;
102  }
103 
104  // get the IF Switch Assembly or the IF Switch Subrack component ID:
105  if (dbObject_mp -> getConfigIFSwitchAssy(configId, componentId, SN)) {
106  target.fkIFSwitch_m = componentId.keyId;
107  }
108 
109  // get the CPDS Assembly or the CPDS Subrack component ID:
110  if (dbObject_mp -> getConfigCPDSAssy(configId, componentId, SN)) {
111  target.fkPowerDist_m = componentId.keyId;
112  }
113 
114  // loop on band number from 1 to 10:
115  CartAssemblyConfig caConfig;
116  bool saveConfig;
117 
118  for (int band = 1; band <= 10; ++band) {
119  caConfig.reset();
120  saveConfig = false;
121 
122  // save the band number as both the port and band for the cartAssembly configuration:
123  caConfig.Id_m = CartAssemblyID(band, band);
124 
125  // load the CCA component for the band:
126  if (dbObject_mp -> getConfigCCA(configId, band, componentId, SN)) {
127  // if found, save its component ID in caConfig.Id_m:
128  caConfig.Id_m.CCAFacility_m = componentId.keyFacility;
129  caConfig.Id_m.CCAId_m = componentId.keyId;
130 
131  // also save it as the ID portion of caConfig.coldCart_m:
132  caConfig.coldCart_m.reset(caConfig.Id_m.CCAFacility_m, caConfig.Id_m.CCAId_m);
133 
134  // if that looks valid, load the associated coldCart configuration:
135  if (caConfig.Id_m.CCAValid()) {
136  if (getColdCartConfig(caConfig.Id_m.CCAFacility_m, caConfig.Id_m.CCAId_m, caConfig.coldCart_m))
137  saveConfig = true; // record that we got at least one valid component for the band.
138  }
139  }
140 
141  // load the WCA component for the band:
142  if (dbObject_mp -> getConfigWCA(configId, band, componentId, SN)) {
143  // if found, save its component ID in caConfig.Id_m:
144  caConfig.Id_m.WCAFacility_m = componentId.keyFacility;
145  caConfig.Id_m.WCAId_m = componentId.keyId;
146 
147  // also save it as the ID portion of caConfig.WCA_m:
148  caConfig.WCA_m.reset(caConfig.Id_m.WCAFacility_m, caConfig.Id_m.WCAId_m);
149 
150  if (caConfig.Id_m.WCAValid()) {
151  // if that looks valid, load the associated WCA configuration:
152  if (getWCAConfig(caConfig.Id_m.WCAFacility_m, caConfig.Id_m.WCAId_m, caConfig.WCA_m))
153  saveConfig = true; // record that we got at least one valid component for the band.
154  }
155  }
156 
157  // if valid configuration was found for this band, apply it to the target FrontEndConfig:
158  if (saveConfig)
159  target.setCartridgeConfig(band, caConfig);
160  }
161  return true;
162 }

198 bool ConfigProviderMySQL::getCartAssemblyConfig(CartAssemblyID CAId, CartAssemblyConfig &target) const {
199 
200  // TODO: this function is duplicated here and in ConfigProviderIniFile. Move to base or refactor away somehow?
201 
202  target.reset();
203 
204  if (!CAId.isValid())
205  return false;
206 
207  target.Id_m = CAId;
208 
209  bool found;
210  found = getColdCartConfig(CAId.CCAFacility_m, CAId.CCAId_m, target.coldCart_m);
211  if (!found)
212  target.coldCart_m.reset();
213 
214  found = getWCAConfig(CAId.WCAFacility_m, CAId.WCAId_m, target.WCA_m);
