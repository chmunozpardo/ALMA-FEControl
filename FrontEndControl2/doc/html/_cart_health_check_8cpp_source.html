239  SLEEP(2500);
240 
241  // If cold, enable LO PA:
242  if (receiverIsCold_m && hasCCA) {
243  if (!ca_m.setLOPowerAmps(true))
244  LOG(LM_ERROR) << context << ": setLOPowerAmps failed for band=" << band << endl;

258  } else
259  done = true;
260  }
261  if (seconds == 99)
262  LOG(LM_INFO) << context << ": timed out waiting for adjustLOPowerAmps for band=" << band << endl;
263  else
264  LOG(LM_INFO) << context << ": enabled and set LO power amps for band=" << band << endl;
265  }
266 
267  // Otherwise, make sure they are off:
268  } else {
269  if (!ca_m.setLOPowerAmps(false))
270  LOG(LM_ERROR) << context << ": setLOPowerAmps failed for band=" << band << endl;
271  else
272  LOG(LM_INFO) << context << ": disabled LO power amps for band=" << band << endl;
273  }
274 
275  // Wait for the cartridge to get all warmed up:
276  LOG(LM_INFO) << context << ": waiting " << warmUpTimeSeconds_m << " seconds for the cartridge to warm up..." << endl;
277  SLEEP(1000 * warmUpTimeSeconds_m);
278 
279  freqLOret = freqLO_m;
280  return true;
